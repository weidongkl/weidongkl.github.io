func (h *Handler) Send(ctx context.Context, cmd Command) error {
	select {
	case h.cmdChan <- cmd:
		select {
		case <-cmd.Done():
			return cmd.Err()
		case <-ctx.Done():
			return ctx.Err()
		}
	case <-ctx.Done():
		return ctx.Err()
	case <-h.stop:
		return errors.New("handler stopped")
	}
}

func (c *ResultCommand[T]) Wait(ctx context.Context) (T, error) {
	select {
	case <-ctx.Done():
		var zero T
		return zero, ctx.Err()
	case res := <-c.resultC:
		return res, c.err
	}
}
